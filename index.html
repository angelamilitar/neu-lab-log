📖 System Documentation

Developer guide and Firebase code snippets for the NEU Lab Usage Log system.

1 Firebase Initialization

Install Firebase SDK and configure your project. Save as firebase.js.

import { initializeApp } from 'firebase/app';
import { getAuth, GoogleAuthProvider } from 'firebase/auth';
import { getFirestore } from 'firebase/firestore';

const firebaseConfig = {
  apiKey: "YOUR_API_KEY",
  authDomain: "neu-lab-log.firebaseapp.com",
  projectId: "neu-lab-log",
};

const app = initializeApp(firebaseConfig);
export const auth = getAuth(app);
export const db = getFirestore(app);
export const googleProvider = new GoogleAuthProvider();

2 Google Auth (NEU Domain Restriction)

async function loginWithGoogle() {
  const result = await signInWithPopup(auth, googleProvider);
  const email = result.user.email;
  if (!email.endsWith('@neu.edu.ph')) {
    await auth.signOut();
    throw new Error('Only @neu.edu.ph accounts allowed');
  }
  const userDoc = await getDoc(doc(db, 'users', result.user.uid));
  if (userDoc.data()?.blocked) throw new Error('Account is blocked');
  return result.user;
}

3 Firestore Structure

// users/{uid}
{ name, email, role: "professor"|"admin", blocked: false }

// rooms/{roomId}
{ roomNumber, building, isOccupied: false, currentLogId: null }

// logs/{logId}
{ professorId, professorName, roomNumber,
  startTime, endTime, college, program,
  section, track, status: "active"|"completed" }

4 Start & End Session

// Start Session
const logRef = await addDoc(collection(db, 'logs'), {
  ...formData, professorId: auth.currentUser.uid,
  startTime: serverTimestamp(), status: 'active'
});
await updateDoc(doc(db, 'rooms', roomId), {
  isOccupied: true, currentLogId: logRef.id
});

// End Session
await updateDoc(doc(db, 'logs', logId), {
  endTime: serverTimestamp(), status: 'completed'
});
await updateDoc(doc(db, 'rooms', roomId), {
  isOccupied: false, currentLogId: null
});
